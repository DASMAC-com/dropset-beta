\begin{algorithm}
\caption{ENTRYPOINT}
\begin{algorithmic}
\INPUT $r_1 = input$ (Input buffer)
\INPUT $r_2 = data$ (Instruction data)
\PROCEDURE{ENTRYPOINT}{$input, data$}
  \STATE $data_l = data.length$
  \STATE $d_d = data.discriminator$
    \IF{$d_d$ == \texttt{Discriminator::RegisterMarket}}
      \STATE \CALL{REGISTER-MARKET}{$input$, $data$, $data_l$}
    \ENDIF
  \STATE abort \texttt{Error::InvalidInstructionDiscriminator}
\ENDPROCEDURE
\end{algorithmic}
\end{algorithm}
