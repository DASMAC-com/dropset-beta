\begin{algorithm}
\caption{REGISTER-MARKET}
\begin{algorithmic}
\INPUT $input$ ($r_1$): Input buffer
\INPUT $instruction$ ($r_2$): Instruction data
\INPUT $instruction\_length$ ($r_3$): Instruction length
\REQUIRE Instruction discriminator is DISCRIMINATOR.REGISTER\_MARKET
\PROCEDURE{REGISTER-MARKET}{$input$, $instruction$, $instruction\_length$}
  \IF{$instruction\_length \neq$ INSTRUCTION\_LENGTH.REGISTER\_MARKET}
    \STATE abort E\_INVALID\_INSTRUCTION\_LENGTH
  \ENDIF
\ENDPROCEDURE
\end{algorithmic}
\end{algorithm}
