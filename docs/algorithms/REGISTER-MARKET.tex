\begin{algorithm}
\caption{REGISTER-MARKET}
\begin{algorithmic}
  \COMMENT{Input buffer.}
  \INPUT $r_1 = input$
  \COMMENT{Instruction data.}
  \INPUT $r_2 = insn$
  \COMMENT{Instruction data length.}
  \INPUT $r_3 = insn\_len$
  \REQUIRE $insn.discriminant$ == \texttt{Discriminant::RegisterMarket}
\PROCEDURE{REGISTER-MARKET}{$input, insn, insn\_len$}
  \IF{$insn\_len$ != \texttt{RegisterMarketData.LEN}}
    \RETURN \texttt{ErrorCode::InvalidInstructionLength}
  \ENDIF
  \IF{$input.n\_accounts$ != \texttt{RegisterMarketAccounts.N\_ACCOUNTS}}
    \RETURN \texttt{ErrorCode::InvalidNumberOfAccounts}
  \ENDIF
\ENDPROCEDURE
\end{algorithmic}
\end{algorithm}
