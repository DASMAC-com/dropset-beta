\begin{algorithm}
\caption{REGISTER-MARKET}
\begin{algorithmic}
  \COMMENT{Pointer to input buffer.}
  \INPUT $r_1$ = input
  \COMMENT{Pointer to instruction data.}
  \INPUT $r_2$ = insn
  \COMMENT{Number of accounts.}
  \INPUT $r_3$ = n\_accounts
  \COMMENT{Instruction data length.}
  \INPUT $r_4$ = insn\_len
  \COMMENT{Pointer to stack frame.}
  \INPUT $r_{10}$ = frame
  \REQUIRE insn.discriminant == \texttt{Discriminant::RegisterMarket}
\PROCEDURE{REGISTER-MARKET}{input, insn, n\_accounts, insn\_len, frame}
  \COMMENT{Validate accounts and initialize frame fields for market PDA creation.}
  \STATE acct = \CALL{MARKET-PDA-PRELUDE}{input, insn, n\_accounts, insn\_len, frame}
  \COMMENT{Derive market PDA, invoke system\_program::CreateAccount.}
  \STATE \CALL{INIT-MARKET-PDA}{input, insn, acct, frame}
  \COMMENT{Validate base token program, initialize base vault.}
  \STATE input\_shifted = \CALL{INIT-BASE-VAULT}{acct, frame}
  \COMMENT{Validate quote token program, initialize quote vault.}
  \STATE \CALL{INIT-QUOTE-VAULT}{acct, frame, input, input\_shifted}
\ENDPROCEDURE
\end{algorithmic}
\end{algorithm}
